\tl_new:N \dailylist_header

{
  \seq_clear:N \dailylist_seq
  \ior_open:Nn \dailylist_stream { #1 }
  \int_zero:N \l_tmpa_int
  % populate the sequence
  \ior_map_inline:Nn \dailylist_stream
  {
    \int_incr:N \l_tmpa_int
    \seq_put_right:Nx \dailylist_seq
    {
     % \int_to_arabic:n { \l_tmpa_int } & \exp_not:n {\fontsize{8}{8}\selectfont ##1 } &
       & \exp_not:n {\fontsize{8}{8}\selectfont ##1 } &
      }
  }
  % build the header
  \tl_set:Nn \dailylist_header_uuno
  {
    \multicolumn{2}{c}{\fontsize{14}{12}\selectfont \bfseries #1} &
    \multicolumn{2}{c}{\cellcolor{RoyalBlue!50}\fontsize{8}{8}\selectfont \bfseries S}  & 
    \multicolumn{3}{c}{\cellcolor{ForestGreen!50}\fontsize{8}{8}\selectfont \bfseries O} \\

     & \textbf{Actividad} &

    \rotatebox{60}{vie,~5~} 
    \\ \hline
  }
  % check the number
  \int_compare:nT { \seq_count:N \dailylist_seq > 181 }
  {
    \__dailylist_halve:
  }
  % make the table
  \par\noindent
  %\begin{table*}[!ht]

}
\cs_new_protected:Nn \__dailylist_halve:
{
  % make it even
  \int_if_odd:nT { \seq_count:N \dailylist_seq }
  {
    \seq_put_right:Nn \dailylist_seq { \multicolumn{5}{c}{} }
  }
  % split in half
  \seq_clear:N \dailylist_temp_seq
  \int_step_inline:nn { \seq_count:N \dailylist_seq / 2 }
  {
    \seq_put_right:Nx \dailylist_temp_seq
    {
      \seq_item:Nn \dailylist_seq { ##1 } &
      \seq_item:Nn \dailylist_seq { ##1 + \seq_count:N
        \dailylist_seq / 2 }
    }
  }
  % replace the sequence
  \seq_set_eq:NN \dailylist_seq \dailylist_temp_seq
  % double the header
  \tl_put_right:Nx \dailylist_header
  {
    & \exp_not:V \dailylist_header
  }
}
