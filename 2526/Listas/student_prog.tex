\cs_new_protected:Nn \__dailylist_halve:
{
  % make it even
  \int_if_odd:nT { \seq_count:N \dailylist_seq }
  {
    \seq_put_right:Nn \dailylist_seq { \multicolumn{25}{c}{} }
  }
  % split in half
  \seq_clear:N \dailylist_temp_seq
  \int_step_inline:nn { \seq_count:N \dailylist_seq / 2 }
  {
    \seq_put_right:Nx \dailylist_temp_seq
    {
      \seq_item:Nn \dailylist_seq { ##1 } &
      \seq_item:Nn \dailylist_seq { ##1 + \seq_count:N
        \dailylist_seq / 2 }
    }
  }
  % replace the sequence
  \seq_set_eq:NN \dailylist_seq \dailylist_temp_seq
  % double the header
  \tl_put_right:Nx \dailylist_header
  {
    & \exp_not:V \dailylist_header
  }
}
