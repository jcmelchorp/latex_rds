\part $\dfrac{4\cdot10^{6}}{-1\cdot10^{4}}=$
\begin{minipage}[c]{.3\textwidth}
    \begin{solutionorbox}[1.5cm]
        \begin{align*}
            \dfrac{4\cdot10^{6}}{-1\cdot10^{4}} & =-4\cdot10^{6-4} \\
                                                & =-4\cdot10^{2}   \\
                                                & =-400
        \end{align*}
    \end{solutionorbox}
\end{minipage}
