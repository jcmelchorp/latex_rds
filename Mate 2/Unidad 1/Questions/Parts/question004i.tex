\part $\left(\dfrac{5}{3}\right)^3=$
\begin{minipage}[c]{.3\textwidth}
    \begin{solutionorbox}[1.5cm]
        \begin{align*}
            \left(\tfrac{5}{3}\right)^3 & = \left(\tfrac{5}{3}\right)\times\left(\tfrac{5}{3}\right)\times\left(\tfrac{5}{3}\right) \\
                                        & = \left(\tfrac{25}{9}\right)\times\left(\tfrac{5}{3}\right)                                                               \\
                                        & = \left(\tfrac{125}{27}\right)
        \end{align*}
    \end{solutionorbox}
\end{minipage}
