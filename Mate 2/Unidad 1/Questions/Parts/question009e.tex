\part $\left(5\cdot10^6\right)\cdot\left(5\cdot10^{-4}\right)=$
\begin{minipage}[c]{.6\textwidth}
    \begin{solutionorbox}[1.5cm]
        \begin{align*}
            \left(5\cdot10^6\right)\cdot\left(5\cdot10^{-4}\right) & =5\cdot5\cdot\left(10^6\cdot10^{-4}\right) \\
                                                                    & =25\cdot10^{6-4}                             \\
                                                                    & =25\cdot10^{2}
        \end{align*}
    \end{solutionorbox}
\end{minipage}
