\part $\left(\dfrac{2}{3}\right)^4=$
 \begin{minipage}[c]{.3\textwidth}
    \begin{solutionorbox}[1.5cm]
        \begin{align*}
            \left(\tfrac{2}{3}\right)^4 & = \left(\tfrac{2}{3}\right)\times\left(\tfrac{2}{3}\right)\times\left(\tfrac{2}{3}\right) \times\left(\tfrac{2}{3}\right)\\
                                        & = \left(\tfrac{4}{9}\right)\times\left(\tfrac{4}{9}\right)                                                               \\
                                        & = \left(\tfrac{16}{81}\right)
        \end{align*}
    \end{solutionorbox}
\end{minipage}
