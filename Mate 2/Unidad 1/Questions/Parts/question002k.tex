\part $1.1\overline{6}=$
\begin{minipage}[c]{.3\textwidth}
    \begin{solutionorbox}[1.5cm]
        Sea \begin{align*}
            x=1.1\overline{6} \Rightarrow & 10x=11.\overline{6}   \\
                                          & 100x=116.\overline{6}
        \end{align*}
        Restando ambas ecuaciones,
        \begin{align*}
            100x    & =116.\overline{6} \\
            -  (10x & =11.\overline{6}) \\
            \cline{1-3}
            90x     & =105              \\
            x       & =\dfrac{105}{90}  \\
            x       & =\dfrac{7}{6}     \\
        \end{align*}
    \end{solutionorbox}
\end{minipage}
