\part $6^5=6^x \cdot 6^4$
\begin{minipage}[c]{.3\textwidth}
    \begin{solutionorbox}[1.5cm]
        \begin{align*}
            6^5           & =6^x\cdot 6^4 \\
            \Rightarrow 5 & =x+4          \\
            x             & =5-4          \\
            x             & = 1
        \end{align*}
    \end{solutionorbox}
\end{minipage}
