\part $s^x=s^5 \cdot s^{-4}$

\begin{minipage}[c]{\linewidth}
    \begin{solutionbox}{4cm}
        \begin{align*}
            s^x & =s^5 \cdot s^{-4} \\
            \Rightarrow x &  =5+(-4)\\
               x & =5-4\\
               x & = 1     
        \end{align*}
    \end{solutionbox}
\end{minipage}
