\part $\left(\dfrac{1}{3}\right)^4=$
\begin{minipage}[c]{.3\textwidth}
    \begin{solutionorbox}[1.5cm]
        \begin{align*}
            \left(\tfrac{1}{3}\right)^4 & = \left(\tfrac{1}{3}\right)\times\left(\tfrac{1}{3}\right)\times\left(\tfrac{1}{3}\right)\times\left(\tfrac{1}{3}\right) \\
                                        & = \left(\tfrac{1}{9}\right)\times\left(\tfrac{1}{9}\right)                                                               \\
                                        & = \times\left(\tfrac{1}{81}\right)
        \end{align*}
    \end{solutionorbox}
\end{minipage}
