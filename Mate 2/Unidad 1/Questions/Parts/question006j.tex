\part $z^7=z^{-2} \cdot z^x$

\begin{minipage}[c]{\linewidth}
    \begin{solutionbox}{4cm}
        \begin{align*}
            z^7           & =z^{-2} \cdot z^x \\
            \Rightarrow 7 & =-2+x             \\
            x             & =7+2              \\
            x             & =9
        \end{align*}
    \end{solutionbox}
\end{minipage}
