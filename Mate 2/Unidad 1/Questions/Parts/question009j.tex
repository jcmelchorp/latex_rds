\part $\dfrac{6\cdot10^{4}}{-3\cdot10^{-2}}=$

\begin{minipage}[t]{\linewidth}
    \begin{solutionbox}{4cm}
        \begin{align*}
            \dfrac{6\cdot10^{4}}{-3\cdot10^{-2}} & =-2\cdot10^{4-(-2)} \\
                                                 & =-2\cdot10^{4+2}             \\
                                                 & =-2 \cdot10^6
        \end{align*}
    \end{solutionbox}
\end{minipage}
