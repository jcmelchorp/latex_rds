\part $\dfrac{9\cdot10^{4}}{-3\cdot10^{-4}}=$
%\begin{minipage}[c]{.6\textwidth}
    \begin{solutionbox}{4cm}
        \begin{align*}
            \dfrac{9\cdot10^{4}}{-3\cdot10^{-4}} & =-3\cdot10^{4-(-4)} \\
                                                 & =-3\cdot10^{4+4}             \\
                                                 & =-3 \cdot10^8
        \end{align*}
    \end{solutionbox}
