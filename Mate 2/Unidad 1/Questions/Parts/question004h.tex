\part $\left(\dfrac{3}{2}\right)^3=$

\begin{minipage}[c]{\linewidth}
    \begin{solutionbox}{3cm}
        \begin{align*}
            \left(\tfrac{3}{2}\right)^3 & = \left(\tfrac{3}{2}\right)\times\left(\tfrac{3}{2}\right)\times\left(\tfrac{3}{2}\right) \\
                                        & = \left(\tfrac{9}{4}\right)\times\left(\tfrac{3}{2}\right)                                                               \\
                                        & = \left(\tfrac{27}{8}\right)
        \end{align*}
    \end{solutionbox}
\end{minipage}
