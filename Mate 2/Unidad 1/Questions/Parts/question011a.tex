\part $\sqrt{\dfrac{225}{196}}=$

\begin{minipage}{\linewidth}
    \begin{solutionbox}{3.5cm}
        \begin{align*}
            \sqrt{\dfrac{225}{196}} & =\dfrac{\sqrt{225}}{\sqrt{196}} \\
                                    & =\dfrac{15}{14}
        \end{align*}
    \end{solutionbox}
\end{minipage}
