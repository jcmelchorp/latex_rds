\part $\sqrt{\dfrac{25}{196}}=$

\begin{minipage}[t]{\linewidth}
    \begin{solutionbox}{3.5cm}
        \begin{align*}
            \sqrt{\dfrac{25}{196}} & =\dfrac{\sqrt{25}}{\sqrt{196}} \\
                                                   & =\dfrac{5}{14} 
        \end{align*}
    \end{solutionbox}
\end{minipage}
