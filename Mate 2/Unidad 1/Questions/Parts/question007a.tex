\part $\dfrac{99^{-5}}{99^{-8}}=$
\begin{minipage}[t]{.7\linewidth}
    \begin{solutionbox}{3cm}
        \begin{align*}
            \dfrac{99^{-5}}{99^{-8}} & =99^{-5-(-8)} \\
                                     & =99^{-5+8}    \\
                                     & =99^{3}
        \end{align*}
    \end{solutionbox}
\end{minipage}
