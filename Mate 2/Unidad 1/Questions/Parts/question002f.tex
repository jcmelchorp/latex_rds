    \begin{solutionorbox}[1.5cm]
        Sea $x=2.\overline{8} \Rightarrow 10x  =28.\overline{8}$.\\
        Restando ambas ecuaciones,
        \begin{align*}
               10x & =28.\overline{8}   \\
            -  (x  & =2.\overline{8}) \\
            \cline{1-3}
              9x  & =26\\
              x  & =\dfrac{26}{9}
        \end{align*}
    \end{solutionorbox}
\end{minipage}
