\part $\left(\dfrac{2}{5}\right)^2=$
\begin{minipage}[c]{.3\textwidth}
    \begin{solutionorbox}[1.5cm]
        \begin{align*}
            \left(\tfrac{2}{5}\right)^2 & = \left(\tfrac{2}{5}\right)\times \left(\tfrac{2}{5}\right)\\ 
               & = \left(\tfrac{4}{25}\right)
        \end{align*}
    \end{solutionorbox}
\end{minipage}
