\part $\left(-3\cdot10^4\right)\cdot\left(2\cdot10^{-3}\right)=$
\begin{minipage}[c]{.6\textwidth}
    \begin{solutionorbox}[1.5cm]
        \begin{align*}
            \left(-3\cdot10^4\right)\cdot\left(2\cdot10^{-3}\right) & =-3\cdot2\cdot\left(10^4\cdot10^{-3}\right) \\
                                                                    & =-6\cdot10^{4-3}                              \\
                                                                    & = -6\cdot10^{1}                               \\
                                                                    & = -60
        \end{align*}
    \end{solutionorbox}
\end{minipage}
