\part $\left(\dfrac{1}{2}\right)^4=$
\begin{minipage}[c]{.3\textwidth}
    \begin{solutionorbox}[1.5cm]
        \begin{align*}
            \left(\tfrac{1}{2}\right)^4 & =\left(\tfrac{1}{2}\right)\times\left(\tfrac{1}{2}\right)\times\left(\tfrac{1}{2}\right)\times\left(\tfrac{1}{2}\right) \\
                                        & = \left(\tfrac{1}{16}\right)
        \end{align*}
    \end{solutionorbox}
\end{minipage}
