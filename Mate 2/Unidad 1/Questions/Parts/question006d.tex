\part $7^8=7^8 \cdot 7^x$
\begin{minipage}[c]{.3\textwidth}
    \begin{solutionorbox}[1.5cm]
        \begin{align*}
            7^8 & =7^8 \cdot 7^x \\
            \Rightarrow 8 &  =8+x\\
               x & =8-8\\
               x & = 0      
        \end{align*}
    \end{solutionorbox}
\end{minipage}
