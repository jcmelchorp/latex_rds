\part $\dfrac{-6\cdot10^{-4}}{-12\cdot10^{-5}}=$
\begin{minipage}[c]{.6\textwidth}
    \begin{solutionorbox}[1.5cm]
        \begin{align*}
            \dfrac{-6\cdot10^{-4}}{-12\cdot10^{-5}} & =0.5\cdot10^{-4-(-5)} \\
                                                   & =0.5\cdot10^{-4+5}    \\
                                                   & =0.5 \cdot10^1        \\
                                                   & = 5
        \end{align*}
    \end{solutionorbox}
\end{minipage}
