\part $\dfrac{-6\cdot10^{-4}}{-12\cdot10^{-5}}=$

\begin{minipage}[t]{.6\linewidth}
    \begin{solutionbox}{4cm}
        \begin{align*}
            \dfrac{-6\cdot10^{-4}}{-12\cdot10^{-5}} & =0.5\cdot10^{-4-(-5)} \\
                                                   & =0.5\cdot10^{-4+5}    \\
                                                   & =0.5 \cdot10^1        \\
                                                   & = 5
        \end{align*}
    \end{solutionbox}
\end{minipage}
