\part $8^7=8^3 \cdot 8^x$
\begin{minipage}[c]{.3\textwidth}
    \begin{solutionorbox}[1.5cm]
        \begin{align*}
            8^7=          & = 8^3 \cdot 8^x \\
            \Rightarrow 7 & =3+x            \\
            x             & =7-3            \\
            x             & =4
        \end{align*}
    \end{solutionorbox}
\end{minipage}
