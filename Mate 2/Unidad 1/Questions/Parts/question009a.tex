\part $\left(9\cdot10^9\right)\cdot\left(-2\cdot10^{-3}\right)=$

\begin{minipage}[t]{\linewidth}
    \begin{solutionbox}{4cm}
        \begin{align*}
            \left(9\cdot10^9\right)\cdot\left(-2\cdot10^{-3}\right) & =9\cdot(-2)\cdot\left(10^9\cdot10^{-3}\right) \\
                                                                    & =-18\cdot10^{9-3}                             \\
                                                                    & = -18\cdot10^{6}                              \\
                                                                    & = -1.8\cdot10^{7}
        \end{align*}
    \end{solutionbox}
\end{minipage}
