\part $\dfrac{-8\cdot10^{7}}{2\cdot10^{6}}=$
\begin{minipage}[c]{.6\textwidth}
    \begin{solutionorbox}[1.5cm]
        \begin{align*}
            \dfrac{-8\cdot10^{7}}{2\cdot10^{6}} & =-4\cdot10^{7-6} \\
                                                   & =-4\cdot10^{1}    \\
                                                   & = -40
        \end{align*}
    \end{solutionorbox}
\end{minipage}
