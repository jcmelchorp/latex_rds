\part $\dfrac{-8\cdot10^{2}}{-2\cdot10^{-3}}=$

\begin{minipage}[t]{\linewidth}
    \begin{solutionbox}{4cm}
        \begin{align*}
            \dfrac{-8\cdot10^{2}}{-2\cdot10^{-3}} & =4\cdot10^{2-(-3)} \\
                                                  & =4\cdot10^{2+3}    \\
                                                  & = 4\cdot10^{5}
        \end{align*}
    \end{solutionbox}
\end{minipage}
