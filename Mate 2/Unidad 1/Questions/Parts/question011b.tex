\part $\sqrt{\dfrac{225}{64}}=$

\begin{minipage}[t]{\linewidth}
    \begin{solutionbox}{3.5cm}
        \begin{align*}
            \sqrt{\dfrac{225}{64}} & =\dfrac{\sqrt{225}}{\sqrt{64}} \\
                                                   & =\dfrac{15}{8}  
        \end{align*}
    \end{solutionbox}
\end{minipage}
