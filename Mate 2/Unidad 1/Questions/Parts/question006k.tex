\part $10^{10}=10^5 \cdot 10^x$

\begin{minipage}[c]{.3\textwidth}
    \begin{solutionorbox}[1.5cm]
        \begin{align*}
            10^{10}        & =10^5 \cdot 10^x \\
            \Rightarrow 10 & =5+x             \\
            x              & =10-5            \\
            x              & =5
        \end{align*}
    \end{solutionorbox}
\end{minipage}
