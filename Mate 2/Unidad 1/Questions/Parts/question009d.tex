\part $\left(2\cdot10^{-6}\right)\cdot\left(6\cdot10^{7}\right)=$
\begin{minipage}[c]{.6\textwidth}
    \begin{solutionorbox}[1.5cm]
        \begin{align*}
            \left(2\cdot10^{-6}\right)\cdot\left(6\cdot10^{7}\right) & =2\cdot6\cdot\left(10^{-6}\cdot10^{7}\right) \\
                                                                    & =12\cdot10^{-6+7}                             \\
                                                                    & = 12\cdot10^{1}                              \\
                                                                    & = 120
        \end{align*}
    \end{solutionorbox}
\end{minipage}
