\part $0.1\overline{6}=$
\begin{minipage}[t]{.7\linewidth}
    \begin{solutionbox}{3cm}
        Sea \begin{align*}
            x=0.1\overline{6} \Rightarrow & 10x=1.\overline{6} \\
            &  100x=16.\overline{6}
        \end{align*}
        Restando ambas ecuaciones,
        \begin{align*}
               100x & =16.\overline{6}   \\
            -  (10x  & =1.\overline{6}) \\
            \cline{1-3}
              90x  & =15\\
              x  & =\dfrac{15}{90} \\
              x  & =\dfrac{1}{6} \\
        \end{align*}
    \end{solutionbox}
\end{minipage}
