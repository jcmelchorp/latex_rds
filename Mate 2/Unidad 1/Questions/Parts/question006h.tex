\part $3^{10}=3^7 \cdot 3^x$
\begin{minipage}[c]{.3\textwidth}
    \begin{solutionorbox}[1.5cm]
        \begin{align*}
            3^{10}         & = 3^7 \cdot 3^x \\
            \Rightarrow 10 & =7+x            \\
            x              & =10-7           \\
            x              & = 3
        \end{align*}
    \end{solutionorbox}
\end{minipage}
