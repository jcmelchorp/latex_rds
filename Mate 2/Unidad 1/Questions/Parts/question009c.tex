\part $\left(-3\cdot10^3\right)\cdot\left(4\cdot10^{6}\right)=$
\begin{minipage}[c]{.6\textwidth}
    \begin{solutionorbox}[1.5cm]
        \begin{align*}
            \left(-3\cdot10^3\right)\cdot\left(4\cdot10^{6}\right) & =-3\cdot4\cdot\left(10^3\cdot10^{6}\right) \\
                                                                    & =-12\cdot10^{3+6}                             \\
                                                                    & = -12\cdot10^{9}                              \\
                                                                    & = -1.2\cdot10^{10}
        \end{align*}
    \end{solutionorbox}
\end{minipage}
