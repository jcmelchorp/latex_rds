\part $2^7=$
\begin{minipage}[c]{.3\textwidth}
    \begin{solutionorbox}[1.5cm]
        \begin{align*}
            2^7 & =2\times2\times2\times2\times2\times2\times2 \\
                & = 4\times4\times4\times2                     \\
                & = 16\times8                                  \\
                & = 128
        \end{align*}
    \end{solutionorbox}
\end{minipage}
