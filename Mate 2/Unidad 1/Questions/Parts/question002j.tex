\part $0.\overline{3}=$
\begin{minipage}[c]{.3\textwidth}
    \begin{solutionorbox}[1.5cm]
        Sea $x=0.\overline{3} \Rightarrow 10x=3.\overline{3}$.\\
        Restando ambas ecuaciones,
        \begin{align*}
               10x & =3.\overline{3}   \\
            -  (x  & =0.\overline{3}) \\
            \cline{1-3}
              9x  & =3\\
              x  & =\dfrac{3}{9} \\
              x  & =\dfrac{1}{3} \\
        \end{align*}
    \end{solutionorbox}
\end{minipage}
