\begin{minipage}[c]{.3\textwidth}
    \begin{solutionorbox}[1.5cm]
        Sea $x=1.\overline{5} \Rightarrow 10x  =15.\overline{5}$.\\
        Restando ambas ecuaciones,
        \begin{align*}
               10x & =15.\overline{5}   \\
            -  (x  & =1.\overline{5}) \\
            \cline{1-3}
              9x  & =14\\
              x  & =\dfrac{14}{9}
        \end{align*}
    \end{solutionorbox}
\end{minipage}
