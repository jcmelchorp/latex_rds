\part $\dfrac{-6\cdot10^{-1}}{-3\cdot10^{-2}}=$
\begin{minipage}[c]{.6\textwidth}
    \begin{solutionorbox}[1.5cm]
        \begin{align*}
            \dfrac{-6\cdot10^{-1}}{-3\cdot10^{-2}} & =2\cdot10^{-1-(-2)} \\
                                                   & =2\cdot10^{-1+2}    \\
                                                   & =2\cdot10^{1}       \\
                                                   & = 20
        \end{align*}
    \end{solutionorbox}
\end{minipage}
