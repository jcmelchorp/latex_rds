\part $0.\overline{2}=$
\begin{minipage}[c]{.3\textwidth}
    \begin{solutionorbox}[1.5cm]
        Sea $x=0.\overline{2} \Rightarrow 10x  =2.\overline{2}$.\\
        Restando ambas ecuaciones,
        \begin{align*}
               10x & =2.\overline{2}   \\
            -  (x  & =0.\overline{2}) \\
            \cline{1-3}
              9x  & =2\\
              x  & =\dfrac{2}{9}
        \end{align*}
    \end{solutionorbox}
\end{minipage}
