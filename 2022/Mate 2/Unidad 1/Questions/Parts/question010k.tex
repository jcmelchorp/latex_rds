\part $\dfrac{4\cdot10^{6}}{-1\cdot10^{4}}=$

\begin{minipage}[t]{\linewidth}
    \begin{solutionbox}{4cm}
        \begin{align*}
            \dfrac{4\cdot10^{6}}{-1\cdot10^{4}} & =-4\cdot10^{6-4} \\
                                                & =-4\cdot10^{2}   \\
                                                & =-400
        \end{align*}
    \end{solutionbox}
\end{minipage}
