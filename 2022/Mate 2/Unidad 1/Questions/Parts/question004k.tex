\part $\left(\dfrac{4}{5}\right)^3=$\begin{minipage}[c]{.3\textwidth}
    \begin{solutionorbox}[1.5cm]
        \begin{align*}
            \left(\tfrac{4}{5}\right)^3 & = \left(\tfrac{4}{5}\right)\times\left(\tfrac{4}{5}\right)\times\left(\tfrac{4}{5}\right) \\
                                        & = \left(\tfrac{16}{25}\right)\times\left(\tfrac{4}{9}\right)                                                               \\
                                        & = \left(\tfrac{64}{225}\right)
        \end{align*}
    \end{solutionorbox}
\end{minipage}
