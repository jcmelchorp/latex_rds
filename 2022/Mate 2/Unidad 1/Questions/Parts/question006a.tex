\part $11^x=11^4 \cdot 11^5$
\begin{minipage}[c]{.3\textwidth}
    \begin{solutionorbox}[1.5cm]
        \begin{align*}
            11^x          & =11^4 \cdot 11^5 \\
            \Rightarrow x & =4+5             \\
            x             & =9
        \end{align*}
    \end{solutionorbox}
\end{minipage}
