\part $\left(-2\cdot10^2\right)\cdot\left(-4\cdot10^{-6}\right)=$

\begin{minipage}[t]{\linewidth}
    \begin{solutionbox}{4cm}
        \begin{align*}
            \left(-2\cdot10^2\right)\cdot\left(-4\cdot10^{-6}\right) & =-2\cdot-4\cdot\left(10^2\cdot10^{-6}\right) \\
                                                                    & =8\cdot10^{2-6}                           \\
                                                                    & = 8\cdot10^{-4}                           
        \end{align*}
    \end{solutionbox}
\end{minipage}
