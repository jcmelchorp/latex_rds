        Restando ambas ecuaciones,
        \begin{align*}
               10x & =16.\overline{6}   \\
            -  (x  & =1.\overline{6}) \\
            \cline{1-3}
              9x  & =15\\
              x  & =\dfrac{15}{9} \\
              x  & =\dfrac{5}{3} \\
        \end{align*}
    \end{solutionorbox}
\end{minipage}
