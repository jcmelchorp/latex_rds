\part $\dfrac{x^{-3}}{x^{-2}}=$

\begin{minipage}[c]{\linewidth}
    \begin{solutionbox}{4cm}
        \begin{align*}
            \frac{x^{-3}}{x^{-2}} & =x^{-3-(-2)} \\
                                   & =x^{-3+2}    \\
                                   & =x^{-1}
        \end{align*}
    \end{solutionbox}
\end{minipage}
