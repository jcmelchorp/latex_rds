\part $\dfrac{4000}{4\cdot10^{-5}}=$
\begin{minipage}[c]{.6\textwidth}
    \begin{solutionorbox}[1.5cm]
        \begin{align*}
            \dfrac{4000}{4\cdot10^{-5}} & =\dfrac{4\cdot10^3}{4\cdot10^{-5}} \\
                                        & = 1\cdot10^{3-(-5)}                \\
                                        & =1\cdot10^{3+5}                    \\
                                        & = 1\cdot10^8
        \end{align*}
    \end{solutionorbox}
\end{minipage}
