\part $-2^6=$

\begin{minipage}[t]{0.9\linewidth}
    \begin{solutionbox}{4.5cm}
        \begin{align*}
            -2^6 & =-2\times2\times2\times2\times2\times2 \\
                 & =-4\times2\times2\times2\times2        \\
                 & =-8\times2\times2\times2               \\
                 & =-16\times2\times2                     \\
                 & =-32\times2                            \\
                 & = -64
        \end{align*}
    \end{solutionbox}
\end{minipage}
