\part $h^4\cdot h^{-5}\cdot h^2=$
\begin{minipage}[c]{.3\textwidth}
    \begin{solutionorbox}[1.5cm]
        \begin{align*}
            h^4\cdot h^{-5}\cdot h^2 & =h^{4-5+2} \\
                            & = h^1 \\
                            & = h
        \end{align*}
    \end{solutionorbox}
\end{minipage}
