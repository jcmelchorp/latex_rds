\part $\dfrac{13^5}{13^{-3}}=$

\begin{minipage}{\linewidth}
    \begin{solutionbox}{3cm}
        \begin{align*}
            \tfrac{13^5}{13^{-3}} & =13^{5-(-3)} \\
                                  & =13^{5+3}    \\
                                  & =13^{8}
        \end{align*}
    \end{solutionbox}
\end{minipage}
