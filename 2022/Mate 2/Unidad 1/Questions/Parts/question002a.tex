\part $0.\overline{1}=$

\begin{minipage}[c]{\linewidth}
    \begin{solutionbox}{6cm}
        Sea $x=0.\overline{1} \Rightarrow 10x  =1.\overline{1}$.\\
        Restando ambas ecuaciones,
        \begin{align*}
               10x & =1.\overline{1}   \\
            -  (x  & =0.\overline{1}) \\
            \cline{1-3}
              9x  & =1\\
              x  & =\dfrac{1}{9}
        \end{align*}
    \end{solutionbox}
\end{minipage}
