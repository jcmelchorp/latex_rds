\part $\left(\dfrac{1}{3}\right)^4=$
 \begin{minipage}[t]{.73\linewidth}
    \begin{solutionbox}{3cm}
        \begin{align*}
            \left(\tfrac{1}{3}\right)^4 & = \tfrac{1}{3}\times\tfrac{1}{3}\times\tfrac{1}{3} \times\tfrac{1}{3}\\
                                        & = \tfrac{1\times1\times1\times1}{3\times3\times3\times3}                                                              \\
                                        & = \tfrac{1}{81}
        \end{align*}
    \end{solutionbox}
\end{minipage}
