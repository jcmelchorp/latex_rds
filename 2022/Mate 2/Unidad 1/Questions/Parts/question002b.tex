\begin{minipage}[c]{\linewidth}
    \begin{solutionbox}{6cm}
        Sea $x=3.\overline{1} \Rightarrow 10x  =31.\overline{1}$.\\
        Restando ambas ecuaciones,
        \begin{align*}
               10x & =31.\overline{1}   \\
            -  (x  & =3.\overline{1}) \\
            \cline{1-3}
              9x  & =28\\
              x  & =\dfrac{28}{9}
        \end{align*}
    \end{solutionbox}
\end{minipage}
