\part $(5x+3)\cdot(7x^2-4x+9)$

\begin{minipage}[c]{0.5\linewidth}
    \begin{solutionbox}{3cm}
        \begin{align*}
            (5x+3)\cdot(7x^2-4x+9) & = 35x^3-20x^2+45x+21x^2-12x+27 \\
                                   & = 35x^3+x^2+33x+27
        \end{align*}
    \end{solutionbox}
\end{minipage}
