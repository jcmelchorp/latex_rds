\begin{minipage}[c]{0.4\linewidth}
  $-\dfrac{3}{4}-\left(-\dfrac{1}{2}\right)=$
\end{minipage}
\begin{minipage}[c]{0.6\linewidth}
  \begin{solutionorbox}[2cm]
    \begin{align*}
        -\tfrac{3}{4}-\left(-\tfrac{1}{2}\right) & = -\tfrac{3}{4}-\left(-\tfrac{2}{4}\right)\\
         & = -\tfrac{3}{4}+\tfrac{2}{4} \\
          & =-\tfrac{1}{4} 
        \end{align*}
\end{solutionorbox}
\end{minipage}
