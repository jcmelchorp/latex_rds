\part $-6-(-6)+7+(-4)+(-1)=$

\begin{minipage}[t]{0.9\linewidth}
  \begin{solutionbox}{4cm}
    \begin{align*}
      -6-(-6)+7+(-4)+(-1) & = -6+6+7-4-1 \\
                          & =  7-4-1     \\
                          & =  7-5       \\
                          & = 2
    \end{align*}
  \end{solutionbox}
\end{minipage}
