\part $\dfrac{9}{5}-\left(-\dfrac{6}{10}\right)=$
\begin{minipage}[c]{0.7\linewidth}
  \begin{solutionorbox}[2cm]
    \begin{align*}
        \frac{9}{5}-\left(-\frac{6}{10}\right) & = \frac{9}{5}-\left(-\frac{3}{5}\right)\\
        & = \frac{9}{5}+\frac{3}{5} \\ 
        & = \frac{12}{5} \\
          & =2.4 
        \end{align*}
\end{solutionorbox}
\end{minipage}
