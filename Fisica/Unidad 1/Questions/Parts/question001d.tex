\part Â¿Cu\'antos segundos hay en un a\~no? (considera que no existen a\~nos bisiestos, es decir, que todos los a\~nos tienen 365 d\'ias).

\begin{minipage}[t]{\linewidth}
  \begin{solutionbox}{4.5cm}
        \begin{align*}
      1 \text{ a\~no} & = 365 \underbrace{\text{ d\'ias}} \\
              & = 365 \times 24 \underbrace{\text{ horas}} \\
              & = 365 \times 24 \times 60 \underbrace{\text{ minutos}} \\
              & = 365 \times 24 \times 60  \times 60 \text{ segundos} \\
              & = 315,705,600 \text{ s}
    \end{align*}
  \end{solutionbox}
\end{minipage}
