\part Â¿Cu\'antas horas hay en un siglo? (considera que no existen a\~nos bisiestos, es decir, que todos los a\~nos tienen 365 d\'ias).

\begin{minipage}[t]{\linewidth}
  \begin{solutionbox}{3.8cm}
        \begin{align*}
      1 \text{ siglo} & = 100 \underbrace{\text{ a\~nos}} \\
              & = 100 \times 365 \underbrace{\text{ d\'ias}} \\
              & = 100 \times 365 \times 24 \text{ horas} \\
              & = 876,000 \text{ horas}
    \end{align*}
  \end{solutionbox}
\end{minipage}
