\part  ¿Cuál es el valor de su velocidad media en la segunda etapa?

\begin{minipage}[c]{\linewidth}
    \begin{solutionbox}{4cm}
 La velocidad media en la segunda etapa del viaje
       (Monterrey, $d=600$ km en $t=4$ h) es:
       \begin{align*}
          v & =\frac{d}{t}                     \\
            & =\frac{600\text{ km}}{4\text{h}} \\
            & =150 \text{ km/h}
       \end{align*}
    \end{solutionbox}
 \end{minipage}
