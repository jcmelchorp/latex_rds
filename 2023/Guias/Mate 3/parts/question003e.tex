\part[10] $y=(x+2)^2-64$

\fullwidth{
    \begin{solutionbox}{5cm}
        \begin{multicols}{2}
            Por despeje:
            \[
                \begin{array}{rl}
                    y=         & (x+2)^2-64                          \\
                               & \text{Ya que $y=0$, entonces }      \\
                    0=         & (x+2)^2-64                          \\
                    64=        & (x+2)^2                             \\
                    \pm 8=     & x+2                                 \\
                    -2 \pm 8=  & x                                   \\
                    \therefore & x_1=-2+8 =6 \text{ y } x_2=-2-8=-10
                \end{array}
            \]
        \end{multicols}
    \end{solutionbox}
}
