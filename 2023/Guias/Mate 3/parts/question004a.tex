\part[15] El área de un rectángulo es 528cm$^2$. Su altura es 1 cm más que el doble del ancho.\\
Sea $a$ el ancho del rectángulo.

\textbf{¿Cuál de las siguientes ecuaciones cuadráticas satisface $a$?}

\begin{oneparchoices}
    \choice $2a^2+a+528=0$
    \CorrectChoice $2a^2+a-528=0$
    \choice $2a^2-a-528=0$
    \choice $2a^2-a+528=0$
\end{oneparchoices}

\begin{solutionbox}{3cm}
    Sea $a$ el ancho del rectángulo, entonces su altura esta dada por $2a+1$, y su área es:
    \[
        \begin{array}{rl}
            a(2a+1)    & =528 \\
            2a^2+a     & =528 \\
            2a^2+a-528 & =0   \\
        \end{array}
    \]
\end{solutionbox}

\textbf{Determina el ancho del rectángulo $a$.}

\begin{solutionbox}{5cm}
    \begin{multicols}{3}
        \[
            \begin{array}{rl}
                2x^2+x-528 & =0    \\
                a          & =2    \\
                b          & =1    \\
                c          & =-528 \\
            \end{array}
        \]
        \vspace{1cm}
        \[
            \begin{array}{rl}
                x_{1,2} & = \dfrac{-b\pm\sqrt{b^2-4ac}}{2a}          \\[1.5em]
                x_{1,2} & = \dfrac{-1\pm\sqrt{1^2-4(2)(-528)}}{2(2)} \\[1.5em]
                x_{1,2} & = \dfrac{-1\pm\sqrt{4225}}{4}              \\[1.5em]
            \end{array}
        \]
        \[
            \begin{array}{rl}
                x_{1,2}        & = \dfrac{-1\pm 65}{4}                   \\[1.5em]
                \therefore x_1 & = \dfrac{-1+65}{4}=\dfrac{64}{4}=16     \\[1.5em]
                x_2            & = \dfrac{-1-65}{4}=\dfrac{-66}{4}=-16.5 \\[1.5em]
            \end{array}
        \]
    \end{multicols}
\end{solutionbox}
