\part[20] $y=(x-2)^2-9$

\fullwidth{
    \begin{solutionbox}{5cm}
        \begin{multicols}{2}
            Por despeje:
            \[
                \begin{array}{rl}
                    y=         & (x-2)^2-9                        \\
                               & \text{Ya que $y=0$, entonces }   \\
                    0=         & (x-2)^2-9                        \\
                    9=         & (x-2)^2                          \\
                    \pm 3=     & x-2                              \\
                    2 \pm 3=   & x                                \\
                    \therefore & x_1=2-3 =-1 \text{ y } x_2=2+3=5
                \end{array}
            \]
        \end{multicols}
    \end{solutionbox}
}
