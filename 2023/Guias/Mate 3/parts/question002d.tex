\part[15] $x^2-11x+18=0$

\begin{solutionbox}{10cm}
    \begin{multicols}{2}
        Por factorización:
        \[
            \begin{array}{rl}
                x^2-11x+18 & = 0                      \\
                (x-9)(x-2) & = 0                      \\
                \therefore & x_1 =2 \text{ y } x_2 =9
            \end{array}
        \]
        \\
        Por fórmula general:
        \[
            \begin{array}{rl}
                y          & =	     x^2-11x+18
                \\
                a          & =1
                \\
                b          & =-11
                \\
                c          & =18
                \\
                x_{1,2}    & = \dfrac{-b\pm\sqrt{b^2-4ac}}{2a}
                \\[2em]
                x_{1,2}    & = \dfrac{-(-11)\pm\sqrt{(-11)^2-4(1)(18)}}{2(1)}
                \\[2em]
                x_{1,2}    & = \dfrac{11\pm\sqrt{49}}{2}
                \\[2em]
                x_{1,2}    & = \dfrac{11\pm 7}{2}
                \\[2em]
                \therefore & x_1 =\dfrac{4}{2}=2 \text{ y }  x_2
                =\dfrac{18}{2}=9                                              \\[2em]
            \end{array}
        \]
    \end{multicols}
\end{solutionbox}
