\part[15] $x^2-5x-14=0$

\begin{solutionbox}{10cm}
    \begin{multicols}{2}
        Por factorización:
        \[
            \begin{array}{rl}
                x^2-5x-14  & = 0                       \\
                (x-7)(x+2) & = 0                       \\
                \therefore & x_1 =7 \text{ y } x_2 =-2
            \end{array}
        \]
        \\
        Por fórmula general:
        \[
            \begin{array}{rl}
                y          & =	      x^2-5x-14
                \\
                a          & =1
                \\
                b          & =-5
                \\
                c          & =-14
                \\
                x_{1,2}    & = \dfrac{-b\pm\sqrt{b^2-4ac}}{2a}
                \\[2em]
                x_{1,2}    & = \dfrac{-(-5)\pm\sqrt{(-5)^2-4(1)(-14)}}{2(1)}
                \\[2em]
                x_{1,2}    & = \dfrac{5\pm\sqrt{81}}{2}
                \\[2em]
                x_{1,2}    & = \dfrac{5\pm 9}{2}
                \\[2em]
                \therefore & x_1 =\dfrac{14}{2}=7 \text{ y }  x_2
                =\dfrac{-4}{2}=-2                                            \\[2em]
            \end{array}
        \]
    \end{multicols}
\end{solutionbox}
