\part[15] El producto de dos enteros positivos es 176. Un número es 5 más que el otro.

Sea $n$ el más pequeño de los números eneteros positivos.\\
\textbf{¿Cuál de las siguientes ecuaciones cuadráticas satisface $n$?}

\begin{oneparchoices}
    \choice $n^2+5a+176=0$
    \CorrectChoice $n^2+5a-176=0$
    \choice $n^2-5a-176=0$
    \choice $n^2-5a+176=0$
\end{oneparchoices}

\begin{solutionbox}{3cm}
    Sea $n$ el menor entero positivo, entonces el segundo número es $n+5$. Entonces, el producto es:
    \[
        \begin{array}{rl}
            n(n+5)     & =176 \\
            n^2+5n     & =176 \\
            n^2+5n-176 & =0
        \end{array}
    \]
\end{solutionbox}
}
\textbf{Encuentra el número más pequeño $n$.}

\begin{solutionbox}{5cm}
    \begin{multicols}{3}
        \[
            \begin{array}{rl}
                n^2+5n-176 & =0    \\
                a          & =1    \\
                b          & =5    \\
                c          & =-176 \\
            \end{array}
        \]
        \vspace{1cm}
        \[
            \begin{array}{rl}
                x_{1,2} & = \dfrac{-b\pm\sqrt{b^2-4ac}}{2a}          \\[1.5em]
                x_{1,2} & = \dfrac{-5\pm\sqrt{5^2-4(1)(-176)}}{2(1)} \\[1.5em]
                x_{1,2} & = \dfrac{-5\pm\sqrt{729}}{2}               \\[1.5em]
            \end{array}
        \]
        \[
            \begin{array}{rl}
                x_{1,2}        & = \dfrac{-5\pm 27}{2}                            \\[1.5em]
                \therefore x_1 & = \dfrac{-5-27}{2}=\dfrac{-32}{2}=-16 \text{ y } \\[1.5em]
                x_2            & = \dfrac{-5+27}{2}=\dfrac{22}{2}=11              \\[1.5em]
            \end{array}
        \]
    \end{multicols}
\end{solutionbox}
}
