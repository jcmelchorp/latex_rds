\part[15] $6x^2+36x+54=0$

\begin{solutionbox}{10cm}
    \begin{multicols}{2}
        Por factorización:
        \[
            \begin{array}{rl}
                6x^2+36x+54 & = 0          \\
                6(x^2+6x+9) & = 0          \\
                6(x+3)(x+3) & = 0          \\
                \therefore  & x_1 = x_2=-3
            \end{array}
        \]
        \\
        Por fórmula general:
        \[
            \begin{array}{rl}
                y          & =	    6x^2+36x+54
                \\
                a          & =6
                \\
                b          & =36
                \\
                c          & =54
                \\
                x_{1,2}    & = \dfrac{-b\pm\sqrt{b^2-4ac}}{2a}
                \\[2em]
                x_{1,2}    & = \dfrac{-(36)\pm\sqrt{36^2-4(6)(54)}}{2(6)}
                \\[2em]
                x_{1,2}    & = \dfrac{-36\pm\sqrt{0}}{12}
                \\[2em]
                x_{1,2}    & = \dfrac{-36\pm 0}{12}
                \\[2em]
                \therefore & x_1 =x_2=\dfrac{-36}{12}=-3                  \\[2em]
            \end{array}
        \]
    \end{multicols}
\end{solutionbox}
}
