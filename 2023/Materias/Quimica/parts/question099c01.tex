\part[2] ¿Qué significa que la rapidez con que se metaboliza un medicamento sea proporcional a la concentración?

\begin{choices}
    \choice Que al disminuir la dosis, la rapidez de metabolización aumenta. 
    \choice Que al disminuir la dosis, la rapidez de metabolización se mantiene. 
    \CorrectChoice Que al aumentar la dosis, la rapidez de metabolización aumenta. 
    \choice Que al aumentar la dosis, la rapidez de metabolización se mantiene.
\end{choices}
