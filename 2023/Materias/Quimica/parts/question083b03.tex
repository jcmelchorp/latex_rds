\part[2] ¿Qué partículas pierden energía cuando chocan con otras partículas distintas?

\begin{choices}
    \choice Las más frías, porque transmiten energía a las otras partículas.
    \choice Las más frías, porque absorben energía de las otras partículas.
    \CorrectChoice Las más calientes, porque transmiten energía a las otras partículas.
    \choice Las más calientes, porque absorben energía de las otras partículas.
\end{choices}
