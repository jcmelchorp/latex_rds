\textbf{¿Cuál de las siguientes fórmulas empíricas corresponde a este compuesto?}

\begin{choices}
    \choice  \ce{CH}
    \choice  \ce{C3H3}
    \choice  \ce{C5H3}
    \choice  \ce{C5H6}
\end{choices}
