Encuentra la soluci\'on a las siguientes ecuaciones.

\begin{multicols}{2}
\begin{parts}
    \part \[3(a+4)=24\]
    \begin{solutionbox}{5.2cm}
        \begin{align*}
            3(a+4)&=24\\
            3a+12&=24\\
            3a&=24-12\\
            3a&=12\\
            a&=\frac{12}{3}\\[0.5em]
            a&=4
        \end{align*}
    \end{solutionbox}
    \part \[-7(x+3)=2(x-9)\]
    \begin{solutionbox}{5cm}
        \begin{align*}
            -7(x+3)&=2(x-9)\\
            -7x-21&=2x-18\\
            -7x-2x&=-18+21\\
            -9x&=3\\
            x&=\frac{3}{-9}\\[0.5em]
            x&=-\frac{1}{3}
        \end{align*}
    \end{solutionbox}
    \part \[\frac{1}{3}x-\frac{1}{6}x+1=0\]
    \begin{solutionbox}{5.2cm}
        \begin{align*}
            \frac{1}{3}x-\frac{1}{6}x+1&=0\\[0.5em]
            \frac{2}{6}x-\frac{1}{6}x&=-1\\[0.5em]
            \frac{1}{6}x&=-1\\[0.5em]
            x&=-1(6)\\
            x&=-6
        \end{align*}
    \end{solutionbox}
    \part \[8(b-2) =-2(b-2)\]
    \begin{solutionbox}{5cm}
        \begin{align*}
            8(b-2)&=-2(b-2)\\
            8b-16&=-2b+4\\
            8b+2b&=4+16\\
            10b&=20\\
            b&=\frac{20}{10}\\[0.5em]
            b&=2
        \end{align*}
    \end{solutionbox}
\end{parts}
\end{multicols}
