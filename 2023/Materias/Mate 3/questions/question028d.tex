Selecciona la opción que concuerde correctamente con los enunciados siguientes:

\begin{multicols}{2}
    \begin{parts}
        \part El cociente de la suma entre la diferencia de dos cantidades.

        \begin{choices}
            \choice $(a + b)(a - b)$
            \choice $(a + b) + (a - b)$
            \choice $(a + b) - (a - b)$
            \CorrectChoice $(a + b) / (a - b)$
        \end{choices}

        \part El doble producto de dos números.

        \begin{choices}
            \choice $-2xy$
            \CorrectChoice $2xy$
            \choice $2x - y$
            \choice $x - 2y$
        \end{choices}

        \part El triple de la diferencia de dos números.

        \begin{choices}
            \choice $3a - b$
            \choice $(a - b)3$
            \choice $a - 3b$
            \CorrectChoice $3(a - b)$
        \end{choices}

        \part La diferencia del triple de dos números.

        \begin{choices}
            \choice $3a - b$
            \choice $3(a - b)$
            \choice $a - 3b$
            \CorrectChoice $3a - 3b$
        \end{choices}

    \end{parts}
\end{multicols}
