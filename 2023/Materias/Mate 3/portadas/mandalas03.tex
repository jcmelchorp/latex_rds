\documentclass[tikz]{standalone}

\begin{document}

\newcounter{fis}
\setcounter{fis}{1}

\begin{tikzpicture}
    \pgfmathsetmacro{\np}{120}
    \pgfmathsetmacro{\as}{360/\np}
    \foreach \x [count=\i from 0] in {0,\as,...,360}{
            \coordinate (\i) at (\x+ \as:3.5cm) {};
        }

    \draw[cyan!50!black] (0) -- (1);
    \draw[cyan!50!black] (1) -- (3);
    \draw[cyan!50!black] (2) -- (5);
    \draw[cyan!50!black] (3) -- (7);
    \foreach \x in {0,1,2,...,\np}{
            \pgfmathsetmacro{\nx}{Mod(\thefis+\x,\np)}
            \stepcounter{fis}
            \draw[cyan!50!black] (\x) -- (\nx);
        }
    \foreach \i in {0,1,...,\np}{
            \draw[black,fill=black] (\i) circle (0.02);
        }
\end{tikzpicture}
\end{document}
