\part[10] $y=(x-7)^2-64$


\begin{solutionbox}{5cm}
    \begin{multicols}{2}
        Por despeje:
        \[
            \begin{array}{rl}
                y=         & (x-7)^2-64                        \\
                           & \text{Ya que $y=0$, entonces }    \\
                0=         & (x-7)^2-64                        \\
                64=        & (x-7)^2                           \\
                \pm 8=     & x-7                               \\
                7 \pm 8=   & x                                 \\
                \therefore & x_1=7-8 =-1 \text{ y } x_2=7+8=15
            \end{array}
        \]
    \end{multicols}
\end{solutionbox}
