\part[10] $y=(x+6)^2-49$


\begin{solutionbox}{5cm}
    \begin{multicols}{2}
        Por despeje:
        \[
            \begin{array}{rl}
                y=         & (x+6)^2-49                          \\
                           & \text{Ya que $y=0$, entonces }      \\
                0=         & (x+6)^2-49                          \\
                49=        & (x+6)^2                             \\
                \pm 7=     & x+6                                 \\
                -6 \pm 7=  & x                                   \\
                \therefore & x_1=-6+7 =1 \text{ y } x_2=-6-7=-13
            \end{array}
        \]
    \end{multicols}
\end{solutionbox}
