Encuentra las soluciones a la siguiente ecuación:
$6x^2+1=487$

\begin{solutionbox}{13cm}
    \begin{multicols}{2}
        Por factorización:
        \begin{align*}
            6x^2+1      & = 487                     \\
            6x^2+1-487  & =0                        \\
            6x^2-486    & = 0                       \\
            6(x^2-81)   & = 0                       \\
            6(x-9)(x+9) & = 0                       \\
            \therefore  & x_1 =-9 \text{ y } x_2 =9
        \end{align*}
        \\
        Por despeje:
        \begin{align*}
            6x^2+1     & = 487                     \\
            6x^2       & = 487-1                   \\
            6x^2       & = 486                     \\
            x^2        & = 81                      \\
            x          & = \pm\sqrt{81}            \\
            x          & = \pm 9                   \\
            \therefore & x_1 =-9 \text{ y } x_2 =9
        \end{align*}
        \\
        Por fórmula general:
        \begin{align*}
            6x^2+1     & = 487                                                       \\
            6x^2+1-487 & = 0                                                         \\
            6x^2-486   & = 0                                                         \\
            a          & =6                                                          \\
            b          & =0                                                          \\
            c          & =-486                                                       \\
            x_{1,2}    & = \dfrac{-b\pm\sqrt{b^2-4ac}}{2a}                           \\[2em]
            x_{1,2}    & = \dfrac{0\pm\sqrt{0^2-4(6)(-486)}}{2(6)}                   \\[2em]
            x_{1,2}    & = \dfrac{\pm\sqrt{11664}}{12}                               \\[2em]
            x_{1,2}    & = \dfrac{\pm108}{6}                                         \\[2em]
            \therefore & x_1 =\dfrac{-108}{12}=-9 \text{ y }  x_2 =\dfrac{108}{12}=9 \\[2em]
        \end{align*}
    \end{multicols}
\end{solutionbox}
