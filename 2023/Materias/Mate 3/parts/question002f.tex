\part[5] $2x^2-16x+14=0$

\begin{solutionbox}{10cm}
    \begin{multicols}{2}
        Por factorización:
        \[
            \begin{array}{rl}
                2x^2-16x+14 & = 0                      \\
                2(x^2-8x+7) & = 0                      \\
                2(x-1)(x-7) & = 0                      \\
                \therefore  & x_1 =1 \text{ y } x_2 =7
            \end{array}
        \]
        \\
        Por fórmula general:
        \[
            \begin{array}{rl}
                y          & =	    2x^2-16x+14
                \\
                a          & =2
                \\
                b          & =-16
                \\
                c          & =14
                \\
                x_{1,2}    & = \dfrac{-b\pm\sqrt{b^2-4ac}}{2a}
                \\[2em]
                x_{1,2}    & = \dfrac{-(-16)\pm\sqrt{(-16)^2-4(2)(14)}}{2(2)}
                \\[2em]
                x_{1,2}    & = \dfrac{16\pm\sqrt{144}}{4}
                \\[2em]
                x_{1,2}    & = \dfrac{16\pm 12}{4}
                \\[2em]
                \therefore & x_1 =\dfrac{4}{4}=1 \text{ y }  x_2
                =\dfrac{28}{4}=7                                              \\[2em]
            \end{array}
        \]
    \end{multicols}
\end{solutionbox}
