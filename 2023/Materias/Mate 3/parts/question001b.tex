\part $3x^2+4=436$

\begin{solutionbox}{10.5cm}
    \begin{multicols}{2}
        Por factorización:
        \[
            \begin{array}{rl}
                3x^2+4            & =             436           \\
                3x^2+4-436        & =        0                  \\
                3x^2-432          & =          0                \\
                3(x^2-144)        & =        0                  \\
                3(x - 12)(x + 12) & =  0                        \\
                \therefore        & x_1 =-12 \text{ y } x_2 =12
            \end{array}
        \]
        \\
        Por despeje:
        \[
            \begin{array}{rl}
                3x^2+4     & = 436                       \\
                3x^2       & = 436-4                     \\
                3x^2       & = 432                       \\
                x^2        & = 144                       \\
                x          & = \pm\sqrt{144}             \\
                x          & = \pm 12                    \\
                \therefore & x_1 =-12 \text{ y } x_2 =12
            \end{array}
        \]
        \\
        Por fórmula general:
        \[
            \begin{array}{rl}
                3x^2+4     & =436                                                      \\
                3x^2+4-436 & = 0                                                       \\
                3x^2-432   & = 0                                                       \\
                a          & =3                                                        \\
                b          & =0                                                        \\
                c          & =432                                                      \\
                x_{1,2}    & = \dfrac{-b\pm\sqrt{b^2-4ac}}{2a}                         \\[2em]
                x_{1,2}    & = \dfrac{0\pm\sqrt{0^2-4(3)(-432)}}{2(3)}                 \\[2em]
                x_{1,2}    & = \dfrac{\pm\sqrt{5184}}{6}                               \\[2em]
                x_{1,2}    & = \dfrac{\pm72}{6}                                        \\[2em]
                \therefore & x_1 =\dfrac{-72}{6}=-12 \text{ y }  x_2 =\dfrac{72}{6}=12 \\[2em]
            \end{array}
        \]
    \end{multicols}
\end{solutionbox}
