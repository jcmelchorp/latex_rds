\part[10] $2x^2-2x-180=0$

\begin{solutionbox}{10cm}
    \begin{multicols}{2}
        Por factorización:
        \[
            \begin{array}{rl}
                2x^2-2x-180  & = 0                        \\
                2(x^2-x-90)  & = 0                        \\
                2(x+9)(x-10) & = 0                        \\
                \therefore   & x_1 =-9 \text{ y } x_2 =10
            \end{array}
        \]
        \\
        Por fórmula general:
        \[
            \begin{array}{rl}
                y          & =	    2x^2-2x-180
                \\
                a          & =2
                \\
                b          & =-2
                \\
                c          & =-180
                \\
                x_{1,2}    & = \dfrac{-b\pm\sqrt{b^2-4ac}}{2a}
                \\[2em]
                x_{1,2}    & = \dfrac{-(-2)\pm\sqrt{(-2)^2-4(2)(-180)}}{2(2)}
                \\[2em]
                x_{1,2}    & = \dfrac{2\pm\sqrt{1444}}{4}
                \\[2em]
                x_{1,2}    & = \dfrac{2\pm 38}{4}
                \\[2em]
                \therefore & x_1 =\dfrac{-36}{4}=-9 \text{ y }  x_2
                =\dfrac{40}{4}=10                                             \\[2em]
            \end{array}
        \]
    \end{multicols}
\end{solutionbox}
