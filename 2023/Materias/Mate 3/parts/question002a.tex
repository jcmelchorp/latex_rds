\part $x^2+x-42=0$


\begin{solutionbox}{10cm}
    \begin{multicols}{2}
        Por factorización:
        \[
            \begin{array}{rl}
                x^2+x-42   & = 0                       \\
                (x+7)(x-6) & = 0                       \\
                \therefore & x_1 =-7 \text{ y } x_2 =6
            \end{array}
        \]
        \\
        Por fórmula general:
        \[
            \begin{array}{rl}
                y          & =	      x^2+x-42
                \\
                a          & =1
                \\
                b          & =1
                \\
                c          & =-42
                \\
                x_{1,2}    & = \dfrac{-b\pm\sqrt{b^2-4ac}}{2a}
                \\[2em]
                x_{1,2}    & = \dfrac{-1\pm\sqrt{1^2-4(1)(-42)}}{2(1)}
                \\[2em]
                x_{1,2}    & = \dfrac{-1\pm\sqrt{169}}{2}
                \\[2em]
                x_{1,2}    & = \dfrac{-1\pm 13}{2}
                \\[2em]
                \therefore & x_1 =\dfrac{12}{2}= \text{ y }  x_2
                =\dfrac{-14}{2}=-7                                     \\[2em]
            \end{array}
        \]
    \end{multicols}
\end{solutionbox}
