\part $y=(x-3)^2-81$


\begin{solutionbox}{5cm}
    \begin{multicols}{2}
        Por despeje:
        \[
            \begin{array}{rl}
                y=         & (x-3)^2-81                        \\
                           & \text{Ya que $y=0$, entonces }    \\
                0=         & (x-3)^2-81                        \\
                81=        & (x-3)^2                           \\
                \pm 9=     & x-3                               \\
                3 \pm 9=   & x                                 \\
                \therefore & x_1=3-9 =-6 \text{ y } x_2=3+9=12
            \end{array}
        \]
    \end{multicols}
\end{solutionbox}
