Encuentra las soluciones de $y=(x-2)^2-9$

\begin{solutionbox}{5cm}
    Por despeje:
    \begin{align*}
        y=         & (x-2)^2-9  \qquad \text{ ya que $y=0$, entonces } \\
        0=         & (x-2)^2-9                                         \\
        9=         & (x-2)^2                                           \\
        \pm 3=     & x-2                                               \\
        2 \pm 3=   & x                                                 \\
        \therefore & x_1=2-3 =-1 \text{ y } x_2=2+3=5
    \end{align*}
\end{solutionbox}
