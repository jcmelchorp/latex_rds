\part[25] $y=-10x^2+490$

\begin{solutionbox}{10.5cm}
    \begin{multicols}{2}
        Por factorización:
        \[
            \begin{array}{rl}
                y          & = -10x^2+490                   \\
                           & \text{Ya que $y=0$, entonces } \\
                0          & = -10(x^2-49)                  \\
                0          & =-10(x-7)(x+7)                 \\
                \therefore & x_1 =-7 \text{ y } x_2 =7
            \end{array}
        \]
        \\
        Por despeje:
        \[
            \begin{array}{rl}
                y             & = -10x^2+490                   \\
                              & \text{Ya que $y=0$, entonces } \\
                0             & = -10x^2+490                   \\
                -490          & = -10x^2                       \\
                490           & = 10x^2                        \\
                49            & = x^2                          \\
                \pm \sqrt{49} & =         x                    \\
                \pm 7         & =         x                    \\
                \therefore    & x_1 =-7 \text{ y } x_2 =7
            \end{array}
        \]
        \\
        Por fórmula general:
        \[
            \begin{array}{rl}
                y          & = -10x^2+490                                                  \\
                a          & =-10                                                          \\
                b          & =0                                                            \\
                c          & =490                                                          \\
                x_{1,2}    & = \dfrac{-b\pm\sqrt{b^2-4ac}}{2a}                             \\[2em]
                x_{1,2}    & = \dfrac{0\pm\sqrt{0^2-4(-10)(490)}}{2(-10)}                  \\[2em]
                x_{1,2}    & = \dfrac{\pm\sqrt{19600}}{-20}                                \\[2em]
                x_{1,2}    & = \dfrac{\pm140}{-20}                                         \\[2em]
                \therefore & x_1 =\dfrac{140}{-20}=-7 \text{ y }  x_2 =\dfrac{-140}{-20}=7 \\[2em]
            \end{array}
        \]
    \end{multicols}
\end{solutionbox}
