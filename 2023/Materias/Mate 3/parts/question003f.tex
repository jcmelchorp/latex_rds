Encuentra las soluciones de $y=(x-1)^2-9$

\begin{solutionbox}{5cm}
    Por despeje:
    \begin{align*}
        y=         & (x-1)^2-9 \qquad \text{ ya que $y=0$, entonces } \\
        0=         & (x-1)^2-9                                        \\
        9=         & (x-1)^2                                          \\
        \pm 3=     & x-1                                              \\
        1 \pm 3=   & x                                                \\
        \therefore & x_1=1-3 =-2 \text{ y } x_2=1+3=4
    \end{align*}
\end{solutionbox}
