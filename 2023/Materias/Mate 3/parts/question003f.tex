\part[10] $y=(x-1)^2-9$


\begin{solutionbox}{5cm}
    \begin{multicols}{2}
        Por despeje:
        \[
            \begin{array}{rl}
                y=         & (x-1)^2-9                        \\
                           & \text{Ya que $y=0$, entonces }   \\
                0=         & (x-1)^2-9                        \\
                9=         & (x-1)^2                          \\
                \pm 3=     & x-1                              \\
                1 \pm 3=   & x                                \\
                \therefore & x_1=1-3 =-2 \text{ y } x_2=1+3=4
            \end{array}
        \]

    \end{multicols}
\end{solutionbox}
