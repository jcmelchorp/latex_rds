\question[10] Una sucesión comienza en 53 y cada término posterior se obtiene al restar 4 en
cada paso.

\begin{parts}
    \part Escribe una expresión algebraica que sea regla de la sucesión.

    \begin{solutionbox}{1.5cm}
    \end{solutionbox}

    \part Analiza las siguientes expresiones. ¿Cuál o cuáles son iguales o equivalentes a la regla de la sucesión que escribieron?

    \begin{checkboxes}
        \choice $53 - 4 (n - 1)$
        \choice $45 - 4(n - 2)$
        \choice $53n - 4$
        \choice $57 - 4n$
    \end{checkboxes}
\end{parts}
