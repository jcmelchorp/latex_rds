\question Se tiene un vaso en forma de prisma recto decagonal con área igual a 25 cm$^2$ y con volumen de 170 cm$^3$.

\begin{parts}
    \part ¿Cuál es su altura? \emph{Describe el método para resolverla}

    \begin{solutionbox}{1.5cm}
        6.8 cm. Se obtiene al dividir el volumen entre el área de la base.
    \end{solutionbox}

    \part Si el lado del decágono es de 2 cm, ¿cuánto es su apotema? Explica su
    obtención.

    \begin{solutionbox}{2cm}
        Como $A =\dfrac{n\times L\times a}{2}$, $a = \dfrac{2A}{n \times L} = \dfrac{2(25)}{10\times 2} = 2.5$ cm.
    \end{solutionbox}
\end{parts}
