Encuentra la soluci\'on a las siguientes ecuaciones.

\begin{multicols}{2}
    \begin{parts}
        \part $5(2x-1)=-25$

        \begin{solutionbox}{5cm}
            \begin{align*}
                5(2x-1) & =-25            \\
                10x-5   & =-25            \\
                10x     & =-25+5          \\
                10x     & =-20            \\
                x       & =\frac{-20}{10} \\[0.5em]
                x       & =-2
            \end{align*}
        \end{solutionbox}


        \part $2(3x-1)=10$

        \begin{solutionbox}{5cm}
            \begin{align*}
                2(3x-1) & =10           \\
                6x-2    & =10           \\
                6x      & =10+2         \\
                6x      & =12           \\
                x       & =\frac{12}{6} \\[0.5em]
                x       & =2
            \end{align*}
        \end{solutionbox}

        \part $x+1=2x-3$

        \begin{solutionbox}{4cm}
            \begin{align*}
                x+1  & =2x-3 \\
                x-2x & =-3-1 \\
                -x   & =-4   \\
                x    & =4
            \end{align*}
        \end{solutionbox}

        \part $5(2x+3)=7x+3$

        \begin{solutionbox}{4cm}
            \begin{align*}
                5(2x+3) & =7x+3 \\
                10x+15  & =7x+3 \\
                10x-7x  & =3-15 \\
                3x      & =-12  \\
                x       & =-4
            \end{align*}
        \end{solutionbox}

        \part $-x+1=3x-4$

        \begin{solutionbox}{5cm}
            \begin{align*}
                -x+1  & =3x-4          \\
                -x-3x & =-4-1          \\
                -4x   & =-5            \\
                x     & =\frac{-5}{-4} \\[0.5em]
                x     & =\frac{5}{4}
            \end{align*}
        \end{solutionbox}

        \part $3(x-2)=2(x+1)$

        \begin{solutionbox}{4cm}
            \begin{align*}
                3(x-2) & =2(x+1) \\
                3x-6   & =2x+2   \\
                3x-2x  & =2+6    \\
                x      & =8
            \end{align*}
        \end{solutionbox}
    \end{parts}
\end{multicols}
