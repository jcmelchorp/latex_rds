Encuentra la soluci\'on a las siguientes ecuaciones.

\begin{multicols}{2}
    \begin{parts}
        \part $4(a+3)=14$

        \begin{solutionbox}{5.2cm}
            \begin{align*}
                4(a+3) & =14          \\
                4a+12  & =14          \\
                4a     & =14-12       \\
                4a     & =2           \\
                a      & =\frac{2}{4} \\[0.5em]
                a      & =\frac{1}{2}
            \end{align*}
        \end{solutionbox}

        \part $-3(x+7)=9(x-1)$

        \begin{solutionbox}{5cm}
            \begin{align*}
                -3(x+7) & =9(x-1)         \\
                -3x-21  & =9x-9           \\
                -3x-9x  & =-9+21          \\
                -12x    & =12             \\
                x       & =\frac{12}{-12} \\[0.5em]
                x       & =-1
            \end{align*}
        \end{solutionbox}

        \part $\frac{1}{2}x-\frac{1}{4}x+1=0$

        \begin{solutionbox}{5.2cm}
            \begin{align*}
                \frac{1}{2}x-\frac{1}{4}x+1 & =0     \\[0.5em]
                \frac{2}{4}x-\frac{1}{4}x   & =-1    \\[0.5em]
                \frac{1}{4}x                & =-1    \\[0.5em]
                x                           & =-1(4) \\
                x                           & =-4
            \end{align*}
        \end{solutionbox}

        \part $2(b-8) =-3(b-3)$

        \begin{solutionbox}{5cm}
            \begin{align*}
                2(b-8) & =-3(b-3)      \\
                2b-16  & =-3b+9        \\
                2b+3b  & =9+16         \\
                5b     & =25           \\
                b      & =\frac{25}{5} \\[0.5em]
                b      & =5
            \end{align*}
        \end{solutionbox}
    \end{parts}
\end{multicols}
