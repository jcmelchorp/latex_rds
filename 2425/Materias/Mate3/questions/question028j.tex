Escribe una expresión algebraica para cada una de los siguientes enunciados:

\begin{multicols}{2}
    \begin{parts}

        \part Ocho menos el cociente de dos y un número.

        \begin{solutionbox}{1.6cm}
            \[8-\dfrac{2}{x}\]
        \end{solutionbox}

        \part Cuatro menos que el cociente de un número y cinco.

        \begin{solutionbox}{1.6cm}
            \[\dfrac{x}{5}-4\]
        \end{solutionbox}

        \part Seis veces la diferencia de cinco y un número.

        \begin{solutionbox}{1.6cm}
            \[6\left(5-x\right)\]
        \end{solutionbox}

        \part Uno más que el cociente de cuatro y un número.

        \begin{solutionbox}{1.6cm}
            \[1+\dfrac{4}{x}\]
        \end{solutionbox}
    \end{parts}
\end{multicols}
