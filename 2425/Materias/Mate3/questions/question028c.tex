Selecciona la opción que concuerde correctamente con los enunciados siguientes:

\begin{multicols}{2}
    \begin{parts}

        Un número disminuido en 3 es 19.

        \begin{choices}
            \choice $\dfrac{x}{3}=19$
            \choice $-3x=19$
            \choice $x^{-3}=19$
            \CorrectChoice $x-3=19$
        \end{choices}

        El cuadrado de un número.

        \begin{choices}
            \CorrectChoice $x^2$
            \choice $x^4$
            \choice $\sqrt{x}$
            \choice $4x^2$
        \end{choices}

        La suma de un número, su doble y su triple es 42.

        \begin{choices}
            \choice $x + y + z = 42$
            \CorrectChoice $x + 2x + 3x = 42$
            \choice $x + x/2 + x/3 = 42$
            \choice $x + x + 2 + x + 3 = 42$
        \end{choices}

        El producto de la suma de dos números por su diferencia.

        \begin{choices}
            \choice $(x+y) - (x - y)$
            \choice $x(x - y)$
            \CorrectChoice $(x + y)(x - y)$
            \choice $xy$
        \end{choices}

    \end{parts}
\end{multicols}
