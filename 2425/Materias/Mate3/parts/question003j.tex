Encuentra las soluciones de $y=(x+6)^2-49$

\begin{solutionbox}{5cm}
    Por despeje:
    \begin{align*}
        y=         & (x+6)^2-49 \quad \text{ ya que $y=0$, entonces } \\
        0=         & (x+6)^2-49                                       \\
        49=        & (x+6)^2                                          \\
        \pm 7=     & x+6                                              \\
        -6 \pm 7=  & x                                                \\
        \therefore & x_1=-6+7 =1 \text{ y } x_2=-6-7=-13
    \end{align*}
\end{solutionbox}
