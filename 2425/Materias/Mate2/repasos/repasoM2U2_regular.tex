                \part $x^2y^3z^4 \cdot x^5z^4=$
                \begin{solutionbox}{2cm}
                    $x^2y^3z^4 \cdot x^5z^4 = x^7y^3z^8$
                \end{solutionbox}

                \columnbreak

                \part $x^3x^2x^3=$

                \begin{solutionbox}{2cm}

                \subsection{\ifprintanswers{Resta de exponentes}\else{}\fi}
